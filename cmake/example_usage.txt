# Example CMakeLists.txt for using the Realm library

cmake_minimum_required(VERSION 3.10)
project(MyProject C CXX)

# Find the Realm package
find_package(Realm REQUIRED)

# Example 1: Daemon application
add_executable(my_daemon_app daemon_main.c)
target_link_libraries(my_daemon_app PRIVATE Realm::Daemon)

# Example 2: Client application
add_executable(my_client_app client_main.c)
target_link_libraries(my_client_app PRIVATE Realm::Client)

# Example 3: Application using both daemon and client
add_executable(my_full_app main.c)
target_link_libraries(my_full_app PRIVATE Realm::Daemon Realm::Client)
